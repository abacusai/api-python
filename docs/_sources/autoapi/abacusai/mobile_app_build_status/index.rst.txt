   :param mobileAppBuildId: build identifier
   :type mobileAppBuildId: str

   :type hostname: str
   :param requiredInput: The required input for the build
   :type requiredInput: str
   :param providers: Apple provider/team options when awaiting a selection
   :type providers: list
   :param selectionType: Whether the pending selection is for a 'team' or a 'provider'
   :type selectionType: str
   :param phoneNumbers: Trusted phone number options when awaiting a 2FA phone selection
   :type phoneNumbers: list
   :param error: The error message for the build
